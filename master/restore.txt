#!/bin/bash

Product=/home/killer/demo/BillingSystem/master/products.txt

while true
do
    echo ""
    echo "========= STOCK MENU ========="
    echo "1. View Product"
    echo "2. Restore Stock"
    echo "0. Exit"
    echo "=============================="

    read -p "Choose option: " choice

    case $choice in

    1)
        echo "Id | Name | Stock"
        echo "-----------------------------"
        grep "|" $Product | awk -F'|' '{print $1, $2, $4}'
        ;;

    2)
        read -p "Enter Product ID: " pid

        product=$(grep "^$pid|" $Product)

        if [ -z "$product" ]; then
            echo "Invalid Product ID!"
            continue
        fi

        name=$(echo "$product" | cut -d'|' -f2 | xargs)
        price=$(echo "$product" | cut -d'|' -f3 | xargs)
        stock=$(echo "$product" | cut -d'|' -f4 | xargs)

        read -p "Enter Stock to Add: " add_stock

        new_stock=$((stock + add_stock))

        sed -i "s/^$pid|$name|$price|$stock/$pid|$name|$price|$new_stock/" $Product

        echo ""
        echo "$name $add_stock - stock updated successfully!"
        echo "New Stock Count = $new_stock"
        ;;

    0)
        echo "Exiting..."
        break
        ;;

    *)
        echo "Invalid choice! Use 1, 2 or 0"
        ;;
    esac
done
